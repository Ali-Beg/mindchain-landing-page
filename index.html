Code Examples

See how easy it is to work with MindChain

import asyncio
from mindchain import MCP, Agent, AgentConfig

async def main():
    # Initialize the MCP
    mcp = MCP(config={
        'log_level': 'INFO',
        'policies': {
            'allow_external_tools': True,
        }
    })
    
    # Create agent configuration
    config = AgentConfig(
        name="ResearchAssistant",
        description="Helps with research tasks",
        system_prompt="You are a helpful research assistant that provides concise and accurate information."
    )
    
    # Create and register an agent
    agent = Agent(config)
    agent_id = mcp.register_agent(agent)
    
    # Run the agent with MCP supervision
    response = await mcp.supervise_execution(
        agent_id=agent_id,
        task=lambda: agent.run("What are the key benefits of transformer models?")
    )
    print(response)
    
    # Clean up
    mcp.unregister_agent(agent_id)

# Run in an async environment
if __name__ == "__main__":
    asyncio.run(main())
import asyncio
from mindchain import MCP, Agent, AgentConfig, AgentOrchestrator

async def main():
    mcp = MCP()
    orchestrator = AgentOrchestrator(mcp)
    
    # Create and register multiple agents
    agents = []
    agent_ids = []
    
    for role in ["Researcher", "Analyst", "Writer"]:
        agent = Agent(AgentConfig(
            name=role,
            description=f"{role} agent",
            system_prompt=f"You are a {role.lower()} specialized in your domain."
        ))
        agent_id = mcp.register_agent(agent)
        agents.append(agent)
        agent_ids.append(agent_id)
    
    # Create a sequential workflow
    topic = "Artificial Intelligence in Healthcare"
    workflow_id = orchestrator.create_sequential_workflow(
        name="Research and Article Creation",
        description=f"Research and create an article about: {topic}",
        agent_ids=agent_ids,
        prompts=[
            f"Research the topic: {topic}",
            "Analyze this research: {previous_result}",
            "Write an article based on: {previous_result}"
        ]
    )
    
    # Execute the workflow
    results = await orchestrator.execute_workflow(workflow_id)
    
    # Get the final article from the last step
    workflow = orchestrator.get_workflow(workflow_id)
    final_step_id = workflow['steps'][-1]['id']
    article = results[final_step_id]
    
    print(article)
    
    # Clean up
    for agent_id in agent_ids:
        mcp.unregister_agent(agent_id)

if __name__ == "__main__":
    asyncio.run(main())

import asyncio
from mindchain import MCP, Policy, Agent, AgentConfig

# Define a custom resource budget policy
class ResourceBudgetPolicy(Policy):
    def __init__(self, max_tokens=10000, max_cost=2.0):
        super().__init__(name="resource_budget")
        self.max_tokens = max_tokens
        self.max_cost = max_cost
        self.tokens_used = 0
        self.cost_incurred = 0.0
        
    async def check(self, agent, action, context):
        # Check if action would exceed token budget
        estimated_tokens = self.estimate_tokens(action)
        if self.tokens_used + estimated_tokens > self.max_tokens:
            return False, "Token budget exceeded"
            
        # Check if action would exceed cost budget
        estimated_cost = self.estimate_cost(agent.model, estimated_tokens)
        if self.cost_incurred + estimated_cost > self.max_cost:
            return False, "Cost budget exceeded"
            
        return True, "Action approved"
    
    async def update(self, agent, action, result):
        tokens = self.count_tokens(result)
        self.tokens_used += tokens
        self.cost_incurred += self.calculate_cost(agent.model, tokens)
        
    def estimate_tokens(self, action):
        # Simple estimation based on prompt length
        return len(action.prompt) // 4
        
    def estimate_cost(self, model, tokens):
        # Cost estimation based on model and tokens
        rates = {"gpt-4": 0.06, "gpt-3.5-turbo": 0.002}
        rate = rates.get(model, 0.01)
        return (tokens / 1000) * rate
        
    def calculate_cost(self, model, tokens):
        return self.estimate_cost(model, tokens)

async def main():
    # Initialize the MCP with the custom policy
    mcp = MCP(config={"log_level": "INFO"})
    
    # Create and register the budget policy
    budget_policy = ResourceBudgetPolicy(max_tokens=5000, max_cost=1.0)
    mcp.add_policy(budget_policy)
    
    # Create agent configuration
    config = AgentConfig(
        name="ResearchAssistant",
        description="Performs research tasks",
        model="gpt-4",
        system_prompt="You are a helpful research assistant."
    )
    
    # Create and register the agent
    agent = Agent(config)
    agent_id = mcp.register_agent(agent)
    
    # Run the agent with policy enforcement
    response = await mcp.supervise_execution(
        agent_id=agent_id,
        task=lambda: agent.run("Research quantum computing advancements in 2025"),
        enforce_policies=True
    )
    
    if response.status == "success":
        print(response.output)
    else:
        print(f"Task failed: {response.error}")
        
    # Get policy usage statistics
    stats = budget_policy.get_stats()
    print(f"Tokens used: {stats['tokens_used']}")
    print(f"Cost incurred: ${stats['cost_incurred']:.4f}")

if __name__ == "__main__":
    asyncio.run(main())

Get Started

Easy steps to begin your MindChain journey

1

Installation

pip install mindchain
2

Basic Configuration

from mindchain import MindChainConfig

config = MindChainConfig()
config.api_key = "your_api_key"
config.save_local()  # Saves configuration locally
3

Create Your First Agent

from mindchain import Agent, MCP

agent = Agent(
    name="assistant",
    description="General purpose assistant",
    model="gpt-4"
)

mcp = MCP()
mcp.register_agent(agent)
4

Run Your Agent

result = mcp.run(
    agent_name="assistant",
    task="Write a short poem about AI"
)

print(result.output)
